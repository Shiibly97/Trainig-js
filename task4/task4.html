Transform from if statement to switch

If statement
let job = "Manager";
let salary = 0;
if (job === "Manager") {
salary = 8000;
} else if (job === "IT" || job === "Support") {
salary = 6000;
} else if (job === "Developer" || job === "Designer") {
salary = 7000;
} else {
salary = 4000;
}

Switch
let job = "Manager";
let salary = 0;
switch (job) {
case ("Manager"):
salary = 8000;
console.log(`The salary is ${salary}`)
break;
case ("IT" || "Support"):
salary = 6000;
console.log(`The salary is ${salary}`)
break;
case ("Developer" || "Desinger"):
salary = 7000;
console.log(`The salary is ${salary}`)
default:
salary = 4000;
console.log(`The salary is ${salary}`);
}

The result is : The salary is 8000
